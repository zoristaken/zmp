SQLite proposed structure

[Song]
id PK
song_name TEXT
artist_name TEXT
release_year INT
album_name TEXT
remix_by TEXT

(example)
(id: 0, song_name: "Enter Sandman", artist_name: "metallica", release_year: 1991, album_name: "metallica", remix_by: "")
(id: 1, song_name: "Tuvan", artist_name: "gaya", release_year: 2011, album_name: "", remix_by: "andy blueman")
(id: 2, song_name: "Enter Sandman", artist_name: "metallica", release_year: 1991, album_name: "metallica", remix_by: "")
(...)


[Filter]
id PK
name TEXT

(example)
(id: 0, name: "metal")
(id: 1, name: "oldschool")
(id: 2, name: "gym")
(...)


[Song_Filter]
id PK
Song.id FK
Filter.id FK

(example)
(id: 0, song_id: 0, filter_id: 0) -> Enter Sandman - Metallica - Metallica - 1991 (metal)
(id: 1, song_id: 0, filter_id: 1) -> Enter Sandman - Metallica - Metallica - 1991 (metal, oldschool)
(id: 2, song_id: 0, filter_id: 2) -> Enter Sandman - Metallica - Metallica - 1991 (metal, oldschool, gym)
(id: 3, song_id: 1, filter_id: 1) -> Tuvan - Gaya - Unknown - 2011 (oldschool)
(id: 4, song_id: 1, filter_id: 2) -> Tuvan - Gaya - Unknown - 2011 (oldschool, gym)
(...)


[Settings]
id PK
key TEXT
value TEXT

(example)
(id: 1, key: "music_folder_path", value: "~/Music/")
(id: 2, key: "play_pause_keybind", value: "f")
(id: 3, key: "randomize_keybind", value: "r")
(id: 4, key: "auto_play", value: "true")
(id: 5, key: "current_filters", value: "0,1,2") -> metal, oldschool, gym
(id: 6, key: "dark_mode", value: "true")
(...)
